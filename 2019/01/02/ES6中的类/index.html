ES6дёӯзҡ„зұ»

еҸ‘иЎЁдәҺ2019-01-02|жӣҙж–°дәҺ2021-12-15|еә”з”ЁејҖеҸ‘
|еӯ—ж•°жҖ»и®Ў:686|йҳ…иҜ»ж—¶й•ҝ:3еҲҶй’ҹ|йҳ…иҜ»йҮҸ:|иҜ„и®әж•°:

ES6дёӯж·»еҠ зҡ„classе…ій”®еӯ—е…¶е®һе№¶йқһзңҹжӯЈзҡ„зұ»пјҢиҖҢжҳҜES5з”ЁеҮҪж•°жқҘжЁЎжӢҹзұ»зҡ„иҜӯжі•зі–гҖӮеңЁES6дёӯеҸҜд»Ҙз”ЁеҰӮдёӢзҡ„иҜӯжі•еҲӣе»әдёҖдёӘзұ»пјҡ

class Students {
    constructor(name, age) {
        this.name = name;
        this.age = age;
    }
    sayName() {
        console.log(this.name);
    }
    sayAge() {
        console.log(this.age);
    }
}

let st1 = new Students('sen', 18);
st1.sayName();
st1.sayAge();

дҪҝз”ЁES6зҡ„classиҜӯжі•зі–е’ҢES5иҮӘе®ҡд№үзҡ„зұ»иҝҳжҳҜжңүдәӣеҢәеҲ«зҡ„пјҡ

  • зұ»зҡ„еЈ°жҳҺдёҚдјҡиў«жҸҗеҚҮпјҢзұ»зҡ„е®һдҫӢеҢ–еҸӘиғҪеңЁзұ»зҡ„еЈ°жҳҺд№ӢеҗҺ
  • зұ»еЈ°жҳҺдёӯзҡ„д»Јз ҒеҸӘиғҪиҝҗиЎҢеңЁдёҘж јжЁЎејҸдёӢ
  • зұ»дёӯзҡ„ж–№жі•жҳҜдёҚеҸҜжһҡдёҫзҡ„
  • е®һдҫӢеҢ–зҡ„ж—¶еҖҷеҝ…йЎ»еҠ newе…ій”®еӯ—
  • еңЁж–№жі•еҶ…йғЁдҝ®ж”№зұ»еҗҚдјҡжҠӣеҮәй”ҷиҜҜпјҢдҪҶеҸҜд»ҘеңЁеӨ–йғЁдҝ®ж”№зұ»еҗҚ

дёӢйқўзҡ„дҫӢеӯҗеұ•зӨәдәҶеҰӮдҪ•еңЁеӨ–йғЁдҝ®ж”№зұ»еҗҚпјҡ

class Students {
    constructor(name, age) {
        this.name = name;
        this.age = age;
    }
    sayName() {
        console.log(this.name);
    }
    sayAge() {
        console.log(this.age);
    }
}

let st1 = new Students('sen', 18);
st1.sayName();
st1.sayAge();

let Stu = Students;

let st2 = new Stu('sen', 18);
st2.sayName();
st2.sayAge();

console.log(st1 instanceof Students); //true
console.log(st2 instanceof Students); //true
console.log(st2 instanceof Stu); //true
console.log(st1 instanceof Stu); //true

зұ»дёӯзҡ„и®ҝй—®еҷЁеұһжҖ§

еҸҜд»ҘдҪҝз”Ёgetе’Ңsetе…ій”®еӯ—жқҘе®ҡд№үи®ҝй—®еҷЁеұһжҖ§пјҢдҫӢеҰӮпјҡ

class Students {
    constructor(name) {
        this.name = name;
    }
    get n() {
        return this.name;
    }
    set n(value) {
        this.name = value;
    }
}

let st1 = new Students('sen', 18);
console.log(st1.n);
st1.n = 'wang';
console.log(st1.n);

йңҖи®Ўз®—зҡ„еұһжҖ§еҗҚ

е’Ңе…¶д»–йңҖи®Ўз®—зҡ„еҗҚеӯ—дёҖж ·пјҢйғҪжҳҜеҠ дёҠж–№жӢ¬еҸ·пјҢдҫӢеҰӮпјҡ

let propName = вҖңsayNameвҖқ;

class Students {
    constructor(name) {
        this.name = name;
    }
    [propName]() {
        console.log(this.name);
    }
}

let st1 = new Students('sen');
st1.sayName();

з»ҷзұ»ж·»еҠ й»ҳи®Өзҡ„з”ҹжҲҗеҷЁ

еҸҜд»ҘеҲ©з”ЁSymbolжқҘз»ҷзұ»ж·»еҠ й»ҳи®Өзҡ„з”ҹжҲҗеҷЁпјҢеҰӮпјҡ

class Collection {
    constructor() {
        this.items = [];
    }
    *[Symbol.iterator]() {
        yield *this.items.values();
    }
}

let col = new Collection();
col.items.push(1);
col.items.push(2);
col.items.push(3);
for( let i of col) {
    console.log(i);
}

з»ҷзұ»ж·»еҠ йқҷжҖҒеұһжҖ§/ж–№жі•

зұ»зҡ„йқҷжҖҒж–№жі•/еұһжҖ§еҸӘиғҪйҖҡиҝҮзұ»еҗҚжқҘи®ҝй—®пјҢиҖҢдёҚиғҪйҖҡиҝҮзұ»зҡ„е®һдҫӢжқҘи®ҝй—®пјҢе…·дҪ“еҒҡжі•жҳҜеңЁж–№жі•/еұһжҖ§е®ҡд№үеүҚйқўеҠ дёҠstaticе…ій”®еӯ—пјҢдҫӢеҰӮпјҡ

class Students {
    constructor(name) {
        this.name = name;
    }
    static sayClassName() {
        console.log("Collection");
    }
}

Students.sayClassName();
let stu = new Students('sen');
stu.sayClassName(); // error

зұ»зҡ„з»§жүҝ

ES6еј•е…ҘдәҶextendsе’ҢsuperжқҘе®һзҺ°зұ»зҡ„з»§жүҝпјҢдҫӢеҰӮпјҡ

class Rectangle {
    constructor(length, width) {
        this.length = length;
        this.width = width;
    }
    getArea() {
        return this.length * this.width;
    }
}

class Square extends Rectangle {
    constructor(length) {
        super(length, length);
    }
}

let sq = new Square(5);
console.log(sq.getArea());

дҪҝз”ЁsuperйңҖиҰҒжіЁж„Ҹпјҡ

  • superеҸӘиғҪз”ЁеңЁжҙҫз”ҹзұ»дёӯ
  • еңЁconstructorйҮҢпјҢsuperиҙҹиҙЈеҲқе§ӢеҢ–thisпјҢжүҖд»Ҙеҝ…йЎ»еңЁthisдҪҝз”Ёд№ӢеүҚи°ғз”Ё
